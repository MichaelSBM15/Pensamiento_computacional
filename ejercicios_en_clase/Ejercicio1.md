## PSEUDOCÓDIGO
Algoritmo POBLACIÓN
	población=0
	tiempo_a_leer=2
	segundoss=0
	minutoss=0
	horass=0
	díass=0
	añoss=0
	Escribir "número de población que quieres evaluar"
	Leer "población"
	segundoss=población*tiempo_a_leer=2
	minutoss=segundoss/60
	horass=minutoss/60
	díass=horass/24
	añoss=díass/365
FinAlgoritmo
